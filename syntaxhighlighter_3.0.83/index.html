        private static string CreateSignatureUrl(string url, string id)
        {
            var timestamp = SignatureHelper.CreateTimestamp();
            var nonce = SignatureHelper.CreateRandomNumber();
            var signature = SignatureHelper.CreateSignature(new string[] { ConfigProperty.WebSite_Token, timestamp, nonce });
            var result = string.Format("{0}?id={1}&signature={2}×tamp={3}&nonce={4}", url, id, signature, timestamp, nonce);
            return result;
        }
